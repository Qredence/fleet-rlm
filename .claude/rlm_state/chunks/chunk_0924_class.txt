class CreditCard(BaseModel):
    number: str = Field(..., min_length=13, max_length=19)
    expiry: str
    cvv: str = Field(..., min_length=3, max_length=4)

    @validator('number')
    def luhn_check(cls, v):
        # Luhn algorithm validation
        if not luhn_valid(v):
