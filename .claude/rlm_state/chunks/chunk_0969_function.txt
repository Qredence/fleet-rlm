def get_console() -> Console:
    """Get the singleton Console instance.

    Creates the instance on first call and reuses it
    for all subsequent calls.

    Returns:
        The shared Console instance.
    """
    global _console
    if _console is None:
        from rich.console import Console
        _console = Console()
    return _console
