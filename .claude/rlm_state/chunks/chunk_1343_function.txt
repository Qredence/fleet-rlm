def _filter_pickleable(d: dict[str, Any]) -> tuple[dict[str, Any], list[str]]:
    kept: dict[str, Any] = {}
    dropped: list[str] = []
    for k, v in d.items():
        if _is_pickleable(v):
            kept[k] = v
        else:
            dropped.append(k)
    return kept, dropped
