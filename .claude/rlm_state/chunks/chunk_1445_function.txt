def count_calls(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        wrapper.num_calls += 1
        print(f"Call {wrapper.num_calls} of {func.__name__}")
        return func(*args, **kwargs)
    wrapper.num_calls = 0
    return wrapper
